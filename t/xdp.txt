# These tests are for tracing XDP programs in the kernel.

# install XDP program
name: xdp::xdp_install
tag: xdp,fnarg
test: -k generic_xdp_install --output-arg 'xdp->prog->aux->id' --output-arg 'xdp->prog->aux->name' --output-arg 'xdp->prog->aux->used_maps[0]->map_type' --output-arg 'xdp->prog->aux->used_maps[0]->name' --output-arg 'dev->ifindex' --output-arg 'dev->name'
match: (array(char[16]))'xdp->prog->aux->name'="crc"
trigger: ./xdpcrc -d lo

---
